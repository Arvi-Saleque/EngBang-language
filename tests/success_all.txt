// Function definitions
add(fx -> shonkha, fy -> shonkha) -> shonkha {
  ferao fx + fy |
}

multiply(fa2 -> shonkha, fb2 -> shonkha) -> shonkha {
  ferao fa2 * fb2 |
}

negate(fz -> shonkha) -> shonkha {
  ferao 0 - fz |
}

double(fn -> shonkha) -> shonkha {
  ferao fn * 2 |
}

square(fn2 -> shonkha) -> shonkha {
  ferao fn2 * fn2 |
}

// Function with conditional inside
max2(ma -> shonkha, mb -> shonkha) -> shonkha {
  ma -> mb {
    [{>=}] {
      ferao ma |
    }
  }
  [] {
    ferao mb |
  }
  |
}

sum3(sa -> shonkha, sb -> shonkha, sc -> shonkha) -> shonkha {
  ferao sa + sb + sc |
}

isHigh(mh -> shonkha) -> shonkha {
  mh -> 80 {
    [{>=}] {
      ferao 1 |
    }
  }
  [] {
    ferao 0 |
  }
  |
}
